48  void run_server::execute()
49  {
50  MTINFO( mtlog, "Creating server objects" );
51 
52  set_status( k_starting );
53 
54  // device manager
55  device_manager t_dev_mgr;
56 
57  // configuration manager
58  config_manager t_config_mgr( f_config, &t_dev_mgr );
59 
60  f_component_mutex.lock();
61 
62  // run database and queue condition
63  acq_request_db t_acq_request_db( &t_config_mgr, f_exe_name );
64  f_acq_request_db = &t_acq_request_db;
65 
66  // server worker
67  server_worker t_worker( &t_dev_mgr, &t_acq_request_db );
68  f_server_worker = &t_worker;
69 
70  // request receiver
71  request_receiver t_receiver( this, &t_config_mgr, &t_acq_request_db, &t_worker, f_exe_name );
72  f_request_receiver = &t_receiver;
73 
74  f_component_mutex.unlock();
75 
76  MTINFO( mtlog, "Starting threads" );
77 
78  thread t_receiver_thread( &t_receiver );
79  thread t_worker_thread( &t_worker );
80 
81  signal_handler t_sig_hand;
82  t_sig_hand.push_thread( &t_receiver_thread );
83  t_sig_hand.push_thread( &t_worker_thread );
84 
85  t_receiver_thread.start();
86  t_worker_thread.start();
87 
88  set_status( k_running );
89  MTINFO( mtlog, "running..." );
90 
91  t_receiver_thread.join();
92  t_worker_thread.join();
93 
94  set_status( k_done );
95 
96  f_component_mutex.lock();
97  f_request_receiver = NULL;
98  f_server_worker = NULL;
99  f_acq_request_db = NULL;
100  f_component_mutex.unlock();
101 
102  if( ! t_sig_hand.got_exit_signal() )
103  {
104  t_sig_hand.pop_thread(); // worker thread
105  t_sig_hand.pop_thread(); // receiver thread
106  }
107 
108  MTINFO( mtlog, "Threads stopped" );
109 
110  f_return = RETURN_SUCCESS;
111 
112  return;
113  }
