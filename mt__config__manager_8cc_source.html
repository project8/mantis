59  bool config_manager::handle_set_request( const param_node& a_msg_payload, const std::string& a_mantis_routing_key, request_reply_package& a_pkg )
60  {
61  string t_routing_key = a_mantis_routing_key;
62  if( t_routing_key.empty() )
63  {
64  a_pkg.send_reply( R_AMQP_ERROR_ROUTINGKEY_NOTFOUND, "No routing key was provided" );
65  return false;
66  }
67 
68  MTDEBUG( mtlog, "Applying a setting" );
69  // apply a configuration setting
70  // the destination node should specify the configuration to set
71  try
72  {
73  f_msc_mutex.lock();
74  t_routing_key += string( "=" ) + a_msg_payload.array_at( "values" )->get_value( 0 );
75  parsable t_routing_key_node_with_value( t_routing_key );
76  MTDEBUG( mtlog, "Parsed routing key and added value:\n" << t_routing_key_node_with_value );
77  if( ! f_master_server_config.node_at( "acq" )->has_subset( t_routing_key_node_with_value ) )
78  {
79  f_msc_mutex.unlock();
80  a_pkg.send_reply( R_DEVICE_ERROR, "Value not found: " + t_routing_key );
81  return false;
82  }
83  f_master_server_config.node_at( "acq" )->merge( t_routing_key_node_with_value );
84  f_msc_mutex.unlock();
85  }
86  catch( exception& e )
87  {
88  f_msc_mutex.unlock();
89  a_pkg.send_reply( R_MESSAGE_ERROR_BAD_PAYLOAD, string( "Invalid payload: " ) + e.what() );
90  return false;
91  }
92 
93  f_msc_mutex.lock();
94  a_pkg.f_reply_node.node_at( "content" )->add( "master-config", *f_master_server_config.node_at( "acq" ) );
95  f_msc_mutex.unlock();
96 
97  return a_pkg.send_reply( R_SUCCESS, "Request succeeded" );
98  }

101  bool config_manager::handle_add_request( const param_node& a_msg_payload, const std::string& a_mantis_routing_key, request_reply_package& a_pkg )
102  {
103  // add something to the master config
104 
105  parsable t_dest_node( a_mantis_routing_key );
106 
107  if( ! t_dest_node[ "add" ].is_node() )
108  {
109  a_pkg.send_reply( R_DEVICE_ERROR, "<add> instruction was not properly formatted" );
110  return false;
111  }
112  if( t_dest_node.node_at( "add" )->has( "device" ) )
113  {
114  MTDEBUG( mtlog, "Attempting to add a device" );
115  // it's expected that any values in the payload are digitizers to be added
116  try
117  {
118  param_node* t_devices_node = f_master_server_config.node_at( "acq" )->node_at( "devices" );
119  for( param_node::const_iterator t_dev_it = a_msg_payload.begin(); t_dev_it != a_msg_payload.end(); ++t_dev_it )
120  {
121  string t_device_type = t_dev_it->first;
122  if( ! t_dev_it->second->is_value() )
123  {
124  MTDEBUG( mtlog, "Skipping <t_device_type> because it's not a value" );
125  continue;
126  }
127  string t_device_name = t_dev_it->second->as_value().as_string();
128 
129  // check if we have a device of this name
130  if( t_devices_node->has( t_device_name ) )
131  {
132  a_pkg.send_reply( R_DEVICE_ERROR, "The master config already has device <" + t_device_name + ">" );
133  return false;
134  }
135 
136  // get the config template from the device manager
137  param_node* t_device_config = f_dev_mgr->get_device_config( t_device_type );
138  if( t_device_config == NULL )
139  {
140  a_pkg.send_reply( R_DEVICE_ERROR, "Did not find device of type <" + t_device_type + ">" );
141  return false;
142  }
143  t_device_config->add( "type", param_value( t_device_type ) );
144  t_device_config->add( "enabled", param_value( 0 ) );
145 
146  // add the configuration to the master config
147  t_devices_node->add( t_device_name, t_device_config );
148  }
149  }
150  catch( exception& e )
151  {
152  a_pkg.send_reply( R_AMQP_ERROR_ROUTINGKEY_NOTFOUND, string( "add.device instruction was not formatted properly: " ) + e.what() );
153  return false;
154  }
155  }
156  else
157  {
158  a_pkg.send_reply( R_AMQP_ERROR_ROUTINGKEY_NOTFOUND, "Invalid add instruction" );
159  return false;
160  }
161 
162  f_msc_mutex.lock();
163  a_pkg.f_reply_node.node_at( "content" )->add( "master-config", f_master_server_config );
164  f_msc_mutex.unlock();
165 
166  return a_pkg.send_reply( R_SUCCESS, "Add request succeeded" );
167  }

169  bool config_manager::handle_remove_request( const param_node& /*a_msg_payload*/, const std::string& a_mantis_routing_key, request_reply_package& a_pkg )
170  {
171  // remove something from the master config
172 
173  parsable t_dest_node( a_mantis_routing_key );
174 
175  if( ! t_dest_node[ "remove" ].is_node() )
176  {
177  a_pkg.send_reply( R_DEVICE_ERROR, "<remove> instruction was not properly formatted" );
178  return false;
179  }
180  if( t_dest_node.node_at( "remove" )->has( "device" ) )
181  {
182  MTDEBUG( mtlog, "Attempting to remove a device" );
183  try
184  {
185  string t_device_name = t_dest_node.node_at( "remove" )->node_at( "device" )->begin()->first;
186 
187  // check if we have a device of this name
188  if( ! f_master_server_config.node_at( "acq" )->node_at( "devices" )->has( t_device_name ) )
189  {
190  a_pkg.send_reply( R_DEVICE_ERROR, "The master config does not have device <" + t_device_name + ">" );
191  return false;
192  }
193 
194  // add the configuration to the master config
195  f_master_server_config.node_at( "acq" )->node_at( "devices" )->erase( t_device_name );
196  }
197  catch( exception& e )
198  {
199  a_pkg.send_reply( R_MESSAGE_ERROR_INVALID_VALUE, string( "remove.device instruction was not formatted properly: " ) + e.what() );
200  return false;
201  }
202  }
203  else
204  {
205  a_pkg.send_reply( R_SUCCESS, "Invalid remove instruction" );
206  return false;
207  }
208 
209  f_msc_mutex.lock();
210  a_pkg.f_reply_node.node_at( "content" )->add( "master-config", f_master_server_config );
211  f_msc_mutex.unlock();
212 
213  return a_pkg.send_reply( R_AMQP_ERROR_ROUTINGKEY_NOTFOUND, "Remove instruction succeeded" );
