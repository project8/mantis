66  bool version::parse( const std::string& a_ver )
67  {
68  if( a_ver == "unknown" )
69  {
70  f_major_ver = 0;
71  f_minor_ver = 0;
72  f_patch_ver = 0;
73  f_version = a_ver;
74  return true;
75  }
76 
77  size_t t_delim_pos_1 = a_ver.find( s_delimeter, 0 );
78  if( t_delim_pos_1 == std::string::npos )
79  {
80  MTERROR( mtlog, "version string <" << a_ver << "> is not in the right format (did not find first delimeter)" );
81  return false;
82  }
83  std::stringstream t_maj_ver_str;
84  t_maj_ver_str << a_ver.substr( 0, t_delim_pos_1 );
85 
86  size_t t_delim_pos_2 = a_ver.find( s_delimeter, t_delim_pos_1 + 1 );
87  if( t_delim_pos_2 == std::string::npos )
88  {
89  MTERROR( mtlog, "version string <" << a_ver << "> is not in the right format (did not find second delimeter)" );
90  return false;
91  }
92  std::stringstream t_min_ver_str;
93  t_min_ver_str << a_ver.substr(t_delim_pos_1 + 1, t_delim_pos_2 );
94 
95  std::stringstream t_patch_ver;
96  t_patch_ver << a_ver.substr( t_delim_pos_2 + 1 );
97 
98  t_maj_ver_str >> f_major_ver;
99  t_min_ver_str >> f_minor_ver;
100  t_patch_ver >> f_patch_ver;
101  f_version = a_ver;
102 
103  return true;
104  }

114  bool version::update_exe_host_user()
115  {
116  const size_t t_bufsize = 1024;
117 
118  // Username
119  char t_username_buf[ t_bufsize ];
120 #ifdef _WIN32
121  DWORD t_bufsize_win = t_bufsize;
122  if( GetUserName( t_username_buf, &t_bufsize_win ) )
123 #else
124  if( getlogin_r( t_username_buf, t_bufsize ) == 0 )
125 #endif
126  {
127  f_username = string( t_username_buf );
128  }
129  else
130  {
131  MTWARN( mtlog, "Unable to get the username" );
132  }
133 
134  // Hostname
135  char t_hostname_buf[ t_bufsize ];
136 #ifdef _WIN32
137  WSADATA wsaData;
138  WSAStartup( MAKEWORD( 2, 2 ), &wsaData );
139 #endif
140  // gethostname is the same on posix and windows
141  if( gethostname( t_hostname_buf, t_bufsize ) == 0 )
142  {
143  f_hostname = string( t_hostname_buf );
144  }
145  else
146  {
147  MTWARN( mtlog, "Unable to get the hostname" );
148  }
149 #ifdef _WIN32
150  WSACleanup();
151 #endif
152 
153  // name of executable
154  //f_exe_name = t_parser.get_value( t_name_exe, f_exe_name );
155 #ifdef _WIN32
156  TCHAR t_exe_buf[ MAX_PATH ];
157  if( ! GetModuleFileName( NULL, t_exe_buf, MAX_PATH ) )
158 #elif __APPLE__
159  char t_exe_buf[ 2048 ];
160  uint32_t t_exe_bufsize = sizeof( t_exe_buf );
161  if( _NSGetExecutablePath( t_exe_buf, &t_exe_bufsize ) != 0 )
162 #elif __linux

164  char t_exe_buf[ t_exe_bufsize ];

166 #endif

168  MTWARN( mtlog, "Could not retrieve executable file name" );
169 #ifdef __APPLE__
170  MTWARN( mtlog, "Executable name buffer is too small; needs size %u\n" << t_bufsize );
171 #endif
172  }
173  f_exe_name = string( t_exe_buf );
174 
175  return true;
176  }
