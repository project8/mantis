293  c = buffer[i];

296

299  case '\n':
300  case '\r':
301  if (!store_comment(p,buffer,pos,i))
302  return KATCP_RESULT_FAIL;

306  }
307  break;

313  return KATCP_RESULT_FAIL;
314  p->state = S_START;
315  pos = i;

317  }
318  break;
319

321  if (!store_setting(p,buffer,pos,i-1))
322  return KATCP_RESULT_FAIL;
323  pos = i;
324  p->state = S_VALUE;
325  break;
326 
327  case S_VALUE:
328  switch(c){
329  case OLABEL:

331  pos = i;
332  break;
333  case VALUE:
334  if (!store_value(p,buffer,pos,i))

336  p->state = S_VALUE;
337  pos = i+1;
338  break;
339  case '\n':

355  break;
356  }
357  break;
358 
359  default:

361  case COMMENT:
362  p->state = S_COMMENT;
363  pos = i;
364  break;
365  case OLABEL:
366  p->state = S_LABEL;
367  pos = i+1;
368  break;
369  case SETTING:
370  p->state = S_SETTING;
371  break;
372  case '\n':
373  case '\r':
374  p->state = S_START;
375  pos = i+1;
376  break;
377  }
378  }
379  }
380

382  fclose(file);
383 
384  return EX_OK;
