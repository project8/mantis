372  bool param_array::has_subset( const param& a_subset ) const
373  {
374  if( ! a_subset.is_array() ) return false;
375  const param_array& t_subset_array = a_subset.as_array();
376  if( t_subset_array.size() > f_contents.size() ) return false;
377  const_iterator t_this_it = f_contents.begin();
378  const_iterator t_that_it = t_subset_array.begin();
379  while( t_that_it != t_subset_array.end() ) // loop condition is on a_subset because it's smaller or equal to this
380  {
381  if( ! (*t_this_it)->has_subset( **t_that_it ) ) return false;
382  ++t_this_it;
383  ++t_that_it;
384  }
385  return true;

466  void param_node::merge( const param_node& a_object )
467  {
468  //MTDEBUG( mtlog, "merging object with " << a_object.size() << " items:\n" << a_object );
469  for( const_iterator it = a_object.f_contents.begin(); it != a_object.f_contents.end(); ++it )
470  {
471  if( ! has( it->first ) )
472  {
473  //MTDEBUG( mtlog, "do not have object <" << it->first << "> = <" << *it->second << ">" );
474  add( it->first, *it->second );
475  continue;
476  }
477  param& t_param = (*this)[ it->first ];
478  if( t_param.is_value() )
479  {
480  //MTDEBUG( mtlog, "replacing the value of \"" << it->first << "\" <" << get_value( it->first ) << "> with <" << *it->second << ">" );
481  replace( it->first, *it->second );
482  continue;
483  }
484  if( t_param.is_node() && it->second->is_node() )
485  {
486  //MTDEBUG( mtlog, "merging nodes")
487  t_param.as_node().merge( it->second->as_node() );
488  continue;
489  }
490  if( t_param.is_array() && it->second->is_array() )
491  {
492  //MTDEBUG( mtlog, "appending array" );
493  t_param.as_array().append( it->second->as_array() );
494  continue;
495  }
496  //MTDEBUG( mtlog, "generic replace" );
497  this->replace( it->first, *it->second );
498  }
