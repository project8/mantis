209 int read_katcl(struct katcl_line *l)
210 {
211  int rr;
212  char *ptr;
213  struct katcl_parse *p;
214 
215  sane_line_katcl(l);
216 
217 #if DEBUG > 1
218  fprintf(stderr, "line: invoking read on line %p\n", l);
219 #endif
220 
221 #ifdef KATCP_CONSISTENCY_CHECKS
222  if(l->l_next == NULL){
223  fprintf(stderr, "line: logic failure, l_next should always be valid\n");
224  abort();
225  }
226 #endif
227 
228  p = l->l_next;
229 
230  if(p->p_size <= p->p_have){
231  ptr = (char *)realloc(p->p_buffer, p->p_size + KATCL_BUFFER_INC);
232  if(ptr == NULL){
233 #ifdef DEBUG
234  fprintf(stderr, "read: realloc to %d failed\n", p->p_size + KATCL_BUFFER_INC);
235 #endif
236  l->l_error = ENOMEM;
237  return -1;
238  }
239 
240  p->p_buffer = ptr;
241  p->p_size += KATCL_BUFFER_INC;
242  }
243 
244  rr = read(l->l_fd, p->p_buffer + p->p_have, p->p_size - p->p_have);
245  if(rr < 0){
246  switch(errno){
247  case ECONNRESET :
248 #ifdef DEBUG
249  fprintf(stderr, "read: connect reset\n");
250 #endif
251  return 1;
252  case EAGAIN :
253  case EINTR :
254  return 0; /* more to be read */
255  default : /* serious error */
256  l->l_error = errno;
257 #ifdef DEBUG
258  fprintf(stderr, "read: read error: %s\n", strerror(errno));
259 #endif
260  return -1;
261  }
262  }
263 
264  if(rr == 0){ /* EOF */
265 #ifdef DEBUG
266  fprintf(stderr, "read: end of file\n");
267 #endif
268  return 1;
269  }
270 
271  p->p_have += rr;
272  return 0;

673 int vsend_katcl(struct katcl_line *l, va_list args)
674 {
675  int flags, result, check;
676  char *string;
677  void *buffer;
678  unsigned long value;
679  int len;
680 #ifdef KATCP_USE_FLOATS
681  double dvalue;
682 #endif
683 
684  check = KATCP_FLAG_FIRST;
685 
686  do{
687  flags = va_arg(args, int);
688  if((check & flags) != check){
689  /* WARNING: tests first arg for FLAG_FIRST */
690 #ifdef DEBUG
691  fprintf(stderr, "vsend: flag check failed 0x%x\n", flags);
692 #endif
693  return -1;
694  }
695  check = 0;
696 
697  switch(flags & KATCP_TYPE_FLAGS){
698  case KATCP_FLAG_STRING :
699  string = va_arg(args, char *);
700  result = append_string_katcl(l, flags & KATCP_ORDER_FLAGS, string);
701  break;
702  case KATCP_FLAG_SLONG :
703  value = va_arg(args, unsigned long);
704  result = append_signed_long_katcl(l, flags & KATCP_ORDER_FLAGS, value);
705  break;
706  case KATCP_FLAG_ULONG :
707  value = va_arg(args, unsigned long);
708  result = append_unsigned_long_katcl(l, flags & KATCP_ORDER_FLAGS, value);
709  break;
710  case KATCP_FLAG_XLONG :
711  value = va_arg(args, unsigned long);
712  result = append_hex_long_katcl(l, flags & KATCP_ORDER_FLAGS, value);
713  break;
714  case KATCP_FLAG_BUFFER :
715  buffer = va_arg(args, void *);
716  len = va_arg(args, int);
717  result = append_buffer_katcl(l, flags & KATCP_ORDER_FLAGS, buffer, len);
718  break;
719 #ifdef KATCP_USE_FLOATS
720  case KATCP_FLAG_DOUBLE :
721  dvalue = va_arg(args, double);
722  result = append_double_katcl(l, flags & KATCP_ORDER_FLAGS, dvalue);
723  break;
724 #endif
725  default :
726 #ifdef DEBUG
727  fprintf(stderr, "vsend: bad type flag 0x%x\n", flags);
728 #endif
729  result = (-1);
730  break;
731  }
732 #if DEBUG > 1
733  fprintf(stderr, "vsend: appended: flags=0x%02x, result=%d\n", flags, result);
734 #endif
735  if(result <= 0){
736 #ifdef DEBUG
737  fprintf(stderr, "vsend: bad result for type 0x%x\n", flags);
738 #endif
739  return -1;
740  }
741  } while(!(flags & KATCP_FLAG_LAST));
742 
743  return 0;
744 }
