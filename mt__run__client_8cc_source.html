53  std::string t_routing_key( f_config.get_value( "dest", "mantis" ) );
54  f_config.erase( "dest" );
55 
56  std::string t_lockout_key_str( f_config.get_value( "key", "" ) );
57  f_config.erase( "key" );
58  bool t_lk_valid = true;
59  uuid_t t_lockout_key = uuid_from_string( t_lockout_key_str, t_lk_valid );
60  if( ! t_lk_valid )

62  MTERROR( mtlog, "Invalid lockout key provided: <" << t_lockout_key_str << ">" );
63  f_return = RETURN_ERROR;
64  return;
65  }
66 
67  param_node t_save_node;
68  if( f_config.has( "save" ) )
69  {
70  t_save_node = *(f_config.node_at( "save" ));
71  }

73 
74  // now all that remains in f_config should be values to pass to the server as arguments to the request
75 
76  msg_request* t_request = NULL;
77  if( t_request_type == "run" )
78  {
79  t_request = create_run_request( t_routing_key );
80  }
81  else if( t_request_type == "get" )
82  {
83  t_request = create_get_request( t_routing_key );
84  }
85  else if( t_request_type == "set" )
86  {
87  t_request = create_set_request( t_routing_key );
88  }

91  t_request = create_cmd_request( t_routing_key );
92  }
93  else

95  MTERROR( mtlog, "Unknown or missing request type: " << t_request_type );
96  f_return = RETURN_ERROR;
97  return;
98  }
99 
100  if( t_request == NULL )
101  {
102  MTERROR( mtlog, "Unable to create request" );
103  f_return = RETURN_ERROR;
104  return;
105  }
106 
107  t_request->set_lockout_key( t_lockout_key );
108 
109  MTDEBUG( mtlog, "Sending message w/ msgop = " << t_request->get_message_op() );
110 
111  std::string t_consumer_tag; // for the reply queue
112  // do_publish will declare the reply queue, start consuming on it, and then publish the request
113  t_request->do_publish( f_channel, f_exchange, t_consumer_tag );
114 
115  if( ! t_consumer_tag.empty() ) // this indicates that the reply queue was created, and we've started consuming on it; we should wait for a reply
116  {
117  MTINFO( mtlog, "Waiting for a reply from the server; use ctrl-c to cancel" );
118 
119  // blocking call to wait for incoming message
120  AmqpClient::Envelope::ptr_t t_envelope = f_channel->BasicConsumeMessage( t_consumer_tag );
121 
122  MTINFO( mtlog, "Response received" );
123 
124  param_node* t_msg_node = NULL;
125  if( t_envelope->Message()->ContentEncoding() == "application/json" )
126  {
127  t_msg_node = param_input_json::read_string( t_envelope->Message()->Body() );
128  }
129  else
130  {
131  MTERROR( mtlog, "Unable to parse message with content type <" << t_envelope->Message()->ContentEncoding() << ">" );
132  }
133 
134  MTINFO( mtlog, "Response from Mantis:\n" <<
135  "Return code: " << t_msg_node->get_value< int >( "retcode", -1 ) << '\n' <<
136  "Return message: " << t_msg_node->get_value( "return_msg", "" ) << '\n' <<
137  *t_msg_node->node_at( "payload" ) );
138 
139  // optionally save "master-config" from the response
140  if( t_save_node.size() != 0 )
141  {
142  if( t_save_node.has( "json" ) )
143  {
144  string t_save_filename( t_save_node.get_value( "json" ) );
145  const param_node* t_master_config_node = t_msg_node->node_at( "payload" );
146  if( t_master_config_node == NULL )

148  MTERROR( mtlog, "Payload is not present" );
149  }
150  else
151  {

153  }
154  }
155  else
156  {
157  MTERROR( mtlog, "Save instruction did not contain a valid file type");
158  }
159 
160  }
161 
162  delete t_msg_node;
163  }
164 
165  f_return = RETURN_SUCCESS;
166 
167  return;

227  msg_request* run_client::create_cmd_request( const std::string& a_routing_key )
228  {
229  param_node* t_payload_node = new param_node();
230 
231  // for the load instruction, the instruction node should be replaced by the contents of the file specified
232  if( f_config.has( "load" ) )
233  {
234  if( ! f_config.node_at( "load" )->has( "json" ) )
235  {
236  MTERROR( mtlog, "Load instruction did not contain a valid file type");
237  delete t_payload_node;
238  return NULL;
239  }
240 
241  string t_load_filename( f_config.node_at( "load" )->get_value( "json" ) );
242  param_node* t_node_from_file = param_input_json::read_file( t_load_filename );
243  if( t_node_from_file == NULL )
244  {
245  MTERROR( mtlog, "Unable to read JSON file <" << t_load_filename << ">" );
246  delete t_payload_node;
247  return NULL;
248  }
249 
250  t_payload_node->merge( *t_node_from_file );
251  f_config.erase( "load" );
252  }
253 
254  // at this point, all that remains in f_config should be other options that we want to add to the payload node
255  t_payload_node->merge( f_config ); // copy f_config
256 
257  return msg_request::create( t_payload_node, OP_CMD, a_routing_key, "", message::k_json );
258  }
