203 struct katcl_parse *remove_index_queue_katcl(struct katcl_queue *q, unsigned int index)
204 {
205  unsigned int end;
206  struct katcl_parse *p;
207 
208  /* WARNING: removing from queue does not decrease reference count */
209 
210  if(q->q_count <= 0){
211 #ifdef DEBUG
212  fprintf(stderr, "parse remove: nothing to remove\n");
213 #endif
214  return NULL;
215  }
216 
217 #ifdef KATCP_CONSISTENCY_CHECKS
218  if(index >= q->q_size){
219  fprintf(stderr, "index %u out of range %u\n", index, q->q_size);
220  abort();
221  }
222  if(q->q_queue[index] == NULL){
223  fprintf(stderr, "index %u (head=%u,count=%u) already null\n", index, q->q_head, q->q_count);
224  abort();
225  }
226 #endif
227 
228 #if DEBUG > 1
229  fprintf(stderr, "queue del[%d]=%p\n", index, q->q_queue[index]);
230 #endif
231 
232  p = q->q_queue[index];
233  q->q_queue[index] = NULL;
234 
235  if(index == q->q_head){
236  /* hopefully the common, simple case: only one interested party */
237  q->q_head = (q->q_head + 1) % q->q_size;
238  q->q_count--;
239 
240 #if DEBUG > 1
241  fprintf(stderr, "remove queue: releasing %p with ref %u\n", p, p->p_refs);
242 #endif
243 
244  return p;
245  }
246 
247  if((q->q_head + q->q_count) > q->q_size){ /* wrapping case */
248  if(index >= q->q_head){ /* position before wrap around, move up head */
249  if(index > q->q_head){
250  memmove(&(q->q_queue[q->q_head + 1]), &(q->q_queue[q->q_head]), (index - q->q_head) * sizeof(struct katcl_parse *));
251  }
252  q->q_queue[q->q_head] = NULL;
253  q->q_head = (q->q_head + 1) % q->q_size;
254  q->q_count--;
255 #if DEBUG > 1
256  fprintf(stderr, "remove queue: releasing %p with ref %u\n", p, p->p_refs);
257 #endif
258  return p; /* WARNING: done here */
259  }
260  } else { /* if no wrapping, we can not be before head */
261  if(index < q->q_head){
262  return NULL;
263  }
264  }
265 
266  /* now move back end by one, to overwrite position at index */
267  end = q->q_head + q->q_count - 1; /* WARNING: relies on count+head never being zero, hence earlier test */
268  if(index > end){
269  return NULL;
270  }

274 
275  fprintf(stderr, "queue remove[%d]=%p\n", index, q->q_queue[index]);
276

279
