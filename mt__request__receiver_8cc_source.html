102  f_channel->DeclareQueue( f_queue_name, false, false, true, true );

104 
105  f_consumer_tag = f_channel->BasicConsume( f_queue_name, f_queue_name, true, false ); // route name used for the queue and the consumer tag; second bool is setting no_ack to false
106  }
107  catch( AmqpClient::AmqpException& e )
108  {
109  MTERROR( mtlog, "AMQP exception caught: " << e.what() );
110  //cancel();
111  f_run_server->quit_server();

113  }

115  {
116  MTERROR( mtlog, "Standard exception caught: " << e.what() );
117  //cancel();
118  f_run_server->quit_server();
119  return;
120  }
121 
122  while( true )
123  {
124  if( f_canceled.load() ) return;
125 
126  // blocking call to wait for incoming message
127  f_status.store( k_listening );
128  MTINFO( mtlog, "Waiting for incoming message" );
129  AmqpClient::Envelope::ptr_t t_envelope = f_channel->BasicConsumeMessage( f_consumer_tag );
130 
131  f_channel->BasicAck( t_envelope );
132  MTINFO( mtlog, "Message received" );
133 
134  if (f_canceled.load()) return;
135 
136  f_status.store( k_processing );
137 
138  message* t_message = message::process_envelope( t_envelope, f_queue_name );
139 
140  if( t_message->is_request() )
141  {
142  msg_request* t_request = static_cast< msg_request* >( t_message );
143  request_reply_package t_reply_pkg( t_request, f_channel, t_exchange_name );
144 
145  // the lockout key must be valid
146  if( ! t_request->get_lockout_key_valid() )
147  {
148  t_reply_pkg.send_reply( R_MESSAGE_ERROR_INVALID_KEY, "Lockout key could not be parsed" );
149  MTWARN( mtlog, "Message had an invalid lockout key" );
150  }
151  else
152  {
153  switch( t_request->get_message_op() )

155  case OP_RUN:
156  {
157  do_run_request( t_request, t_reply_pkg );
158  break;
159  }
160  case OP_GET:

164  } // end "get" operation
165  case OP_SET:

168  break;
169  } // end "set" operation

174  }

183  }
184  }

192  MTINFO( mtlog, "Message handled" );
