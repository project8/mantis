67  message* message::process_envelope( amqp_envelope_ptr a_envelope, const std::string& a_queue_name )
68  {
69  param_node* t_msg_node = NULL;
70  encoding t_encoding;
71  if( a_envelope->Message()->ContentEncoding() == "application/json" )
72  {
73  t_encoding = k_json;
74  t_msg_node = param_input_json::read_string( a_envelope->Message()->Body() );
75  }
76  else if( a_envelope->Message()->ContentEncoding() == "application/msgpack" )
77  {
78  t_encoding = k_msgpack;
79  t_msg_node = param_input_msgpack::read_string( a_envelope->Message()->Body() );
80  }
81  else
82  {
83  MTERROR( mtlog, "Unable to parse message with content type <" << a_envelope->Message()->ContentEncoding() << ">" );
84  return NULL;
85  }
86 
87  if( t_msg_node == NULL )
88  {
89  MTERROR( mtlog, "Message body could not be parsed; skipping request" );
90  return NULL;
91  }
92 
93  string t_routing_key = a_envelope->RoutingKey();
94 
95  MTDEBUG( mtlog, "Processing message:\n" <<
96  "Routing key: " << t_routing_key <<
97  *t_msg_node );
98 
99  message* t_message = NULL;
100  switch( t_msg_node->get_value< unsigned >( "msgtype" ) )
101  {
102  case T_REQUEST:
103  {
104  msg_request* t_request = msg_request::create(
105  t_msg_node->node_at( "payload" ),
106  t_msg_node->get_value< unsigned >( "msgop", OP_UNKNOWN ),
107  t_routing_key,
108  a_queue_name,
109  t_encoding);
110  t_request->set_reply_to( a_envelope->Message()->ReplyTo() );
111 
112  bool t_lockout_key_valid = true;
113  t_request->set_lockout_key( uuid_from_string( t_msg_node->get_value( "lockout_key", "" ), t_lockout_key_valid ) );
114  t_request->set_lockout_key_valid( t_lockout_key_valid );
115 
116  t_message = t_request;
117  break;
118  }
119  default:
120  {
121  MTWARN( mtlog, "Message received with unhandled type: " << t_msg_node->get_value< unsigned >( "msgtype" ) );
122  return NULL;
123  break;
124  }
125  }
126 
127  // set message fields
128  t_message->set_correlation_id( a_envelope->Message()->CorrelationId() );
129  t_message->set_timestamp( t_msg_node->get_value( "timestamp", "" ) );
130 
131  t_message->set_sender_info( new param_node( *(t_msg_node->node_at( "sender_info" ) ) ) );
132 
133  if( t_msg_node->has( "payload" ) )
134  {
135  if( (*t_msg_node)[ "payload" ].is_node() )
136  {
137  t_message->set_payload( new param_node( *(t_msg_node->node_at( "payload" ) ) ) );
138  }
139  else
140  {
141  MTWARN( mtlog, "Non-node payload is present; it will be ignored" );
142  t_message->set_payload( new param_node() );
143  }
144  }
145  else
146  {
147  t_message->set_payload( new param_node() );
148  }
149 
150  return t_message;
151  }

170  bool message::encode_message_body( std::string& a_body ) const
171  {
172  param_node t_body_node;
173  t_body_node.add( "msgtype", param_value( get_message_type() ) );
174  t_body_node.add( "timestamp", param_value( get_absolute_time_string() ) );
175  t_body_node.add( "sender_info", new param_node( *f_sender_info ) );
176  t_body_node.add( "payload", f_payload->clone() ); // use a clone of f_payload
177 
178  if( ! this->derived_modify_message_body( t_body_node ) )
179  {
180  MTERROR( mtlog, "Something went wrong in the derived-class modify_body_message function" );
181  return false;
182  }
183 
184  switch( f_encoding )
185  {
186  case k_json:
187  if( ! param_output_json::write_string( t_body_node, a_body, param_output_json::k_compact ) )
188  {
189  MTERROR( mtlog, "Could not convert message body to string" );
190  return false;
191  }
192  return true;
193  break;
194  default:
195  MTERROR( mtlog, "Cannot encode using <" << interpret_encoding() << "> (" << f_encoding << ")" );
196  return false;
197  break;
198  }

200  return false;
