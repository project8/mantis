49  bool device_manager::configure( acq_request& a_acq_request )
50  {

52  {
53  param_node* t_acq_config = a_acq_request.node_at( "acquisition" );
54  if( t_acq_config == NULL )
55  {
56  MTERROR( mtlog, "Acquisition configuration is missing" );
57  return false;
58  }
59 
60  param_node* t_device_config = t_acq_config->node_at( "devices" );
61  if( t_device_config == NULL )
62  {
63  MTERROR( mtlog, "Device configuration is missing" );
64  return false;
65  }
66 
67  // For now: find the first device that's enabled
68  // TODO: for mutli-device usage, will use all enabled devices
69  param_node* t_enabled_dev_config = NULL;
70  param_node::iterator t_node_it;
71  for( t_node_it = t_device_config->begin(); t_node_it != t_device_config->end(); ++t_node_it )
72  {
73  try
74  {
75  if( t_node_it->second->as_node().get_value< bool >( "enabled", false ) )
76  {
77  t_enabled_dev_config = &( t_node_it->second->as_node() );
78  break;
79  }
80  }
81  catch( exception& e )
82  {
83  MTWARN( mtlog, "Found non-node param object in \"devices\"" );
84  }
85  }
86  if( t_enabled_dev_config == NULL )
87  {
88  MTERROR( mtlog, "Did not find an enabled device" );
89  return false;
90  }
91 
92  if( ! set_device( t_node_it->second->as_node().get_value( "type" ) ) )
93  {
94  MTERROR( mtlog, "Unable to set device" );
95  return false;
96  }
97 
98  if( ! f_device->initialize( t_acq_config, t_enabled_dev_config ) )
99  {
100  MTERROR( mtlog, "Unable to configure device" );
101  return false;
102  }
103  }
104  catch( exception& e )

107  }
