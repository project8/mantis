33  param_node* param_input_json::read_file( const std::string& a_filename )
34  {
35  FILE* t_config_file = fopen( a_filename.c_str(), "r" );
36  if( t_config_file == NULL )
37  {
38  MTERROR( mtlog, "file <" << a_filename << "> did not open" );
39  return NULL;
40  }
41  rapidjson::FileStream t_file_stream( t_config_file );
42 
43  rapidjson::Document t_config_doc;
44  if( t_config_doc.ParseStream<0>( t_file_stream ).HasParseError() )
45  {
46  unsigned errorPos = t_config_doc.GetErrorOffset();
47  rewind( t_config_file );
48  unsigned iChar, newlineCount = 1, lastNewlinePos = 0;
49  int thisChar;
50  for( iChar = 0; iChar != errorPos; ++iChar )
51  {
52  thisChar = fgetc( t_config_file );
53  if( thisChar == EOF )
54  {
55  break;
56  }
57  if( thisChar == '\n' || thisChar == '\r' )
58  {
59  newlineCount++;
60  lastNewlinePos = iChar + 1;
61  }
62  }
63  if( iChar == errorPos )
64  {
65  MTERROR( mtlog, "error parsing config file :\n" <<
66  '\t' << t_config_doc.GetParseError() << '\n' <<
67  "\tThe error was reported at line " << newlineCount << ", character " << errorPos - lastNewlinePos );
68  }
69  else
70  {
71  MTERROR( mtlog, "error parsing config file :\n" <<
72  '\t' << t_config_doc.GetParseError() <<
73  "\tend of file reached before error location was found" );
74  }
75  fclose( t_config_file );
76  return NULL;
77  }
78  fclose( t_config_file );
79 
80  return param_input_json::read_document( t_config_doc );
81  }
